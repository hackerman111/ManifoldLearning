\Section{A single-index model}
\label{Ssingind}

Consider the e.d.r. regression problem
\begin{EQA}
	Y_{i}
	&=&
	\fs(\betav^{\T} \Xv_{i}) + \eps_{i} \, ,
\label{YifXieiedrSI}
\end{EQA}
where \( \Xv_{i} \in \R^{\dimd} \) for large \( \dimd \), \( \betav \) is an index vector and
\( \fs \) is a univariate \emph{link function}.
Introduce also a function \( \gs(x) = \E \bigl( Y \cond \Xv=\xv \bigr) \).
A structural single-index assumption yields the relations
\begin{EQA}
	\gs(\xv)
	&=&
	\fs(\betav^{\T} \xv) ,
	\\
	\nabla \gs(\xv)
	&=&
	\fs'(\betav^{\T} \xv) \betav.
\label{fxgKxedrSI}
\end{EQA} 
In particular, the derivative of \( \gs \) at any point \( \xv \) is proportional
to the target vector \( \betav \).
This suggests recovering \( \betav \) from the nonparametrically estimated gradient of \( \gs \).


\Subsection{Preliminaries}
%Define \( \Psi(\xv) = \bigl( \psi_{m}(\xv) \bigr) \in \R^{\dimd+1} \) for \( \psi_{0}(\xv) \equiv 1 \), 
%\(	\psi_{m}(\xv) =	\langle \xv ,\ev_{m} \rangle \), \( m = 1,\ldots,\dimd \). 
For a collection of centers \( \{ \xv_{j} \}_{j \leq J} \),
localizing tensor \( \TEDR \), 
and a localizing kernel \( \Kern \), consider the weights
\begin{EQA}
	\weight_{ij}
	&=&
	\Kern\bigl( \| \TEDR(\Xv_{i} - \xv_{j}) \|^{2} \bigr) 
	\, ,
	\qquad
	i=1,\ldots,n,
	\quad
	j \leq \JJJ
	\, .
\label{s7cuqkdwywdhbjwbfjwbdj}
\end{EQA}
At the first step, we apply \( \TEDR = \hiso^{-2} \Id_{\dimp} \)
leading to isotropic weights 
\( \weight_{ij} = \Kern\bigl( \hiso^{-2} \| \Xv_{i} - \xv_{j} \|^{2} \bigr) \).
The bandwidth \( \hiso \) has to be selected to ensure sufficiently many points in almost every vicinity of \( \xv_{j} \)
of radius \( \hiso \). 
Compute for each \( j \leq \JJJ \)
\begin{EQA}
	\Imav_{j}
	&=&
	\sumi Y_{i} \linv{\Xv_{i} - \xv_{j}} \weight_{ij} \in \R^{\dimd+1} 
	\, ,
\label{ytd7e7ydwuhd5eythdY}
\end{EQA}
and also \( \dimd+1 \)-symmetric matrix
\begin{EQA}
	\KSe_{j}
	&=&
	\sumi \linv{\Xv_{i} - \xv_{j}} \, \linv{\Xv_{i} - \xv_{j}}^{\T} \weight_{ij}
	=
	\begin{pmatrix}
		N_{j} & \Sv_{j}
		\\
		\Sv_{j}^{T} & \VP_{j}
	\end{pmatrix} 
	\, ,
\label{ytd7e7ydwuhd5eythd}
\end{EQA}
where
\begin{EQA}[c]
	N_{j}
	\eqdef
	\sumi \weight_{ij}
	\, ,
	\quad
	\Sv_{j}
	\eqdef
	\sumi (\Xv_{i} - \xv_{j}) \weight_{ij}
	\, ,
	\quad
	\VP_{j}
	\eqdef
	\sumi (\Xv_{i} - \xv_{j})(\Xv_{i} - \xv_{j})^{\T} \weight_{ij}
	\, .
\end{EQA}
Model equation \eqref{YifXieiedrSI} implies
\begin{EQA}
	\Ima_{j}
	&=&
	\sumi Y_{i} \linv{\Xv_{i} - \xv_{j}} \weight_{ij}  
	=
	\sumi \fs(\betav^{\T} \Xv_{i}) \linv{\Xv_{i} - \xv_{j}} \weight_{ij}
	+ \sumi \eps_{i} \linv{\Xv_{i} - \xv_{j}} \weight_{ij} \, .
\label{hgghjkklvddfrdshwey}
\end{EQA}
Further, localization by the weights 
\( \weight_{ij} = \Kern\bigl( \hiso^{-2} \| \Xv_{i} - \xv_{j} \|^{2} \bigr) \) enables us to locally approximate \( \fs(t) \) 
in a vicinity of \( t_{j} = \langle \betav, \xv_{j} \rangle \) by a linear function
\( \fcn_{j} + \fcl_{j} (t - t_{j}) \).
This leads to the approximation 
\begin{EQA}
	&& \nquad
	\sumi  \fs(\betav^{\T} \Xv_{i}) \linv{\Xv_{i} - \xv_{j}} \weight_{ij} 
	\approx 
	\sumi \bigl\{ \fcn_{j} + \fcl_{j} (\Xv_{i} - \xv_{j})^{\T} \betav \bigr\} \linv{\Xv_{i} - \xv_{j}} \weight_{ij} 
	\\
	&=&
	\KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} 
	=
	\begin{pmatrix}
		\fcn_{j} N_{j} + \fcl_{j} \Sv_{j}^{\T} \betav
		\\
		\fcn_{j} \Sv_{j} + \fcl_{j} \VP_{j} \betav
	\end{pmatrix}.
\label{yd6x5ewbduwuqf64wfgfy6}
\end{EQA}
The calming device suggests introducing additional parameters (observables) 
\( \imav_{j} = \E \Imav_{j} \).
The full dimensional parameter vector \( \prmtv \) includes
the target parameter \( \betav \), the observables \( \imav = (\imav_{j}) \in \R^{J(\dimd+1)} \), 
and the coefficients \( \thetav = (\fcn_{j},\fcl_{j})_{j \leq J} \in \R^{2J} \).
The corresponding objective function reads
\begin{EQA}
	\LL(\prmtv)
	&=&
	\LL(\betav,\thetav,\imav)
	=
	\frac{1}{2} \| \Imav - \imav \|^{2} 
	+ \frac{1}{2} \sum_{j} 
		\Bigl\| \imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2} \, .
\label{gtgyuy6ew5dtde32322jt}
\end{EQA}
Due to the product \( \fcl_{j} \betav \), this objective function is not quadratic in the scope of parameters.
More precisely, it is a polynomial of order 4 and it can be considered as a special case of the error-in-operator model.
The challenge for a careful analysis is a lack of identifiability: rescaling the vector \( \betav \) by any constant
\( c \) and all the coefficients \( \fcl_{j} \) by \( c^{-1} \) does not change the model equation. 
To overcome this problem and enforce identifiability, we introduce an additional penalty which push 
\( \betav \) to be as close as possible to a fixed vector \( \betav_{0} \in \R^{\dimd} \)
meaning a starting guess.
One can use \( \betav_{0} = (1,0,\ldots,0)^{\T} \).
It is only important that \( \langle \betav,\ev_{1} \rangle \) significantly deviates from zero.
With a tuning parameter \( \lambda \), this leads to
\begin{EQA}
	\LL_{\lambda}(\prmtv)
	&=&
	\frac{1}{2} \| \Imav - \imav \|^{2} 
	+ \frac{1}{2} \sum_{j} \Bigl\| \imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2} 
	+ \frac{\lambda}{2} \| \betav - \betav_{0} \|^{2} \, .
\label{tjjc7c6f5fd5e5erjwyhfd}
\end{EQA}
%

\paragraph{Elimination of the observables \( \zv_{j} \)}
Representation \eqref{tjjc7c6f5fd5e5erjwyhfd} is useful for theoretical analysis.
Numerically, one can eliminate the auxiliary variables \( \zv = (\zv_{j}) \).

\begin{lemma}
For any \( j \leq \JJJ \) and any
\( \fcn_{j},\fcl_{j} \, , \betav \), it holds
\begin{EQA}[c]
	\inf_{\zv_{j}} \Bigl\{ \| \Imav_{j} - \imav_{j} \|^{2} 
	+ \Bigl\| \imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2} \Bigr\} 
	=
	\frac{1}{2} \Bigl\| \Imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2}
	\, .
\end{EQA}
\end{lemma}

This result reduces problem \eqref{tjjc7c6f5fd5e5erjwyhfd} to
\begin{EQA}[c]
	\sum_{j} \Bigl\| \Imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2} + 2 \lambda \| \betav - \betav_{0} \|^{2}
	\to
	\min_{\betav,(\fcn_{j},\fcl_{j})}
	\, .
\label{ikdfjoiefjoieuvbjmr3e}
\end{EQA}

\Subsection{Minimization of \( \LL_{\lambda}(\prmtv) \) by alternating}
Now we describe the alternating optimization procedure for minimization of \eqref{ikdfjoiefjoieuvbjmr3e}.
The idea is to iterate optimization of the objective function w.r.t. \( (\fcn_{j},\fcl_{j}) \) for \( \betav \) fixed
and w.r.t. \( \betav \) for \( (\fcn_{j},\fcl_{j})_{j \leq \JJJ} \) fixed.
Note that for a fixed \( \betav \), the optimization problem \eqref{ikdfjoiefjoieuvbjmr3e}
can be decomposed into small subproblems 
\begin{EQA}[c]
%	\inf_{(\fcn_{j},\fcl_{j})} 
	\Bigl\| \Imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2}
	\to \min_{(\fcn_{j},\fcl_{j})}
	\, ,
	\qquad
	j \leq \JJJ
	\, .
\end{EQA}
Each of them is quadratic in \( (\fcn_{j},\fcl_{j}) \) and can be solved analytically.

\begin{lemma}
\label{Lcjlj}
It holds with \( \Imav_{j} = (\Ima_{j,0},\Imav_{j,1}) \)
\begin{EQA}[rcl]
%	&& \nquad
	\inf_{\fcn_{j},\fcl_{j}} \Bigl\| \Imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2}
	&=&
	\inf_{\fcn_{j},\fcl_{j}} \bigl\{ 
		| \Ima_{j,0} - \fcn_{j} N_{j} - \fcl_{j} \Sv_{j}^{\T} \betav |^{2}
		+ \| \Imav_{j,1} - \fcn_{j} \Sv_{j} - \fcl_{j} \VP_{j} \betav \|^{2} 
	\bigr\}
	\, .
\end{EQA}
{\color{blue} please, provide a closed-form solution in terms of \( \Ima_{j,0} \), \( N_{j} \), \( \Sv_{j}^{\T} \betav \),
\( \Imav_{j,1}^{\T} \Sv_{j} \), \( \Imav_{j,1}^{\T} \VP_{j} \betav \), \( \| \Sv_{j} \|^{2} \), \( \| \VP_{j} \betav \|^{2} \), 
\( \Sv_{j}^{\T} \VP_{j} \betav \).
Check
\begin{EQA}[c]
	(\fcn_{j},\fcl_{j})
	=
	\begin{pmatrix}
		N_{j}^{2} + \| \Sv_{j} \|^{2} & N_{j} \Sv_{j}^{\T} \betav + \Sv_{j}^{\T} \VP_{j} \betav
		\\
		N_{j} \Sv_{j}^{\T} \betav + \Sv_{j}^{\T} \VP_{j} \betav & | \Sv_{j}^{\T} \betav |^{2} + \| \VP_{j} \betav \|^{2}
	\end{pmatrix}^{-1} \binom{\Ima_{j,0} N_{j} + \Imav_{j,1}^{\T} \Sv_{j}}{\Ima_{j,0} \Sv_{j}^{\T} \betav + \Imav_{j,1}^{\T} \VP_{j} \betav}
	\, .
\end{EQA}
}
\end{lemma}


Similarly, for fixed \( \{ (\fcn_{j},\fcl_{j}), j \leq \JJJ \} \), minimization of \eqref{ikdfjoiefjoieuvbjmr3e} w.r.t. \( \betav \)
is quadratic and can be solved analytically.

\begin{lemma}
\label{Lbeta}
It holds with \( \Imav_{j} = (\Ima_{j,0},\Imav_{j,1}) \)
\begin{EQA}[rcl]
	&& \nquad
	\inf_{\betav} \Bigl\{ \sum_{j} \Bigl\| \Imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav} \Bigr\|^{2}
		+ \lambda \| \betav - \betav_{0} \|^{2} 
	\Bigr\}	
	\\
	&=&
	\inf_{\betav} \sum_{j} \bigl\{ \| \Ima_{j,0} - \fcn_{j} N_{j} - \fcl_{j} \Sv_{j}^{\T} \betav \|^{2}
		+ \| \Imav_{j,1} - \fcn_{j} \Sv_{j} - \fcl_{j} \VP_{j} \betav \|^{2} 
	\bigr\}
	+ \lambda \| \betav - \betav_{0} \|^{2}
	\, .
\end{EQA}
{\color{blue} please, provide a closed-form solution in terms of 
\begin{EQA}[c]
	\sum_{j} \fcl_{j} \Sv_{j} (\Ima_{j,0} - \fcn_{j} N_{j})
	\, ,
	\qquad
	\sum_{j} \fcl_{j} \VP_{j} (\Imav_{j,1} - \fcn_{j} \Sv_{j})
	\, ,
	\qquad
	\sum_{j} \fcl_{j}^{2} \bigl( \Sv_{j} \Sv_{j}^{\T} + \VP_{j}^{2} \bigr)
\end{EQA}
yielding
\begin{EQA}[c]
	\betav
	=
	\biggl\{ \sum_{j} \fcl_{j}^{2} \bigl( \Sv_{j} \Sv_{j}^{\T} + \VP_{j}^{2} \bigr) + \lambda \Id_{\dimp} \biggr\}^{-1}
	\sum_{j} \bigl\{ \fcl_{j} \Sv_{j} (\Ima_{j,0} - \fcn_{j} N_{j})
	+ \fcl_{j} \VP_{j} (\Imav_{j,1} - \fcn_{j} \Sv_{j}) 
	+ \lambda \betav_{0}
	\bigr\}
\end{EQA}
}
\end{lemma}


The alternating minimization can be described as follows.
\begin{mylist}

\item 
Initialize \( \betav^{(0)} \) by a prior information or other procedure like PCA.
Set \( m=1 \).
\item 
With \( \betav^{(m-1)} \) fixed, compute for every \( j \leq \JJJ \)
\begin{EQA}[c]
	(\fcn_{j}^{(m)},\fcl_{j}^{(m)})
	=
	\inf_{(\fcn_{j},\fcl_{j})} \Bigl\| \Imav_{j} - \KSe_{j} \binom{\fcn_{j}}{\fcl_{j} \, \betav^{(m-1)}} \Bigr\|^{2}
\end{EQA}
as suggested in Lemma~\ref{Lcjlj}.
\item 
Then with these \( (\fcn_{j}^{(m)},\fcl_{j}^{(m)}) \), update \( \betav^{(m)} \):
\begin{EQA}[c]
	\betav^{(m)}
	=
	\inf_{\betav} \Bigl\{ \sum_{j} \Bigl\| \Imav_{j} - \KSe_{j} \binom{\fcn_{j}^{(m)}}{\fcl_{j}^{(m)} \, \betav} \Bigr\|^{2}
		+ \lambda \| \betav - \betav^{(m-1)} \|^{2} 
	\Bigr\}	
	\, ,
\end{EQA}
as suggested in Lemma~\ref{Lbeta}.
\item 
Iterate several time for \( m=1,2,\ldots, \) until convergence.
\end{mylist}

\Subsection{Anisotropic weights and local tensor \( \TEDR \)}
The local weights \( \weight_{ij} \) from \eqref{s7cuqkdwywdhbjwbfjwbdj} involve a local tensor \( \TEDR \).
The single-index model structure \( \gs(\xv) = \fs(\betav^{\T} \xv) \) in \eqref{YifXieiedrSI} suggests
anisotropic weights
\begin{EQA}
	\weight_{ij}
	&=&
	\Kern\bigl( h^{-2} |(\Xv_{i} - \xv_{j})^{\T} \betavs|^{2} \bigr) .
\label{s7cuqkdwywdhbjwbfjwbdji}
\end{EQA}
Unfortunately, such defined weights require knowledge of the index vector \( \betavs \) and 
a correctly specified single-index model. 
We follow the structure-adaptive approach from \cite{HJPS}, which applies anisotropic weights 
with an increasing anisotropy parameter during iterations.
As mentioned earlier, for initialization, we apply a non-informative choice 
\begin{EQA}[c]
	\weight_{ij} = \Kern\bigl( \hiso_{0}^{-2} \| \Xv_{i} - \xv_{j} \|^{2} \bigr) 
	\, .
\end{EQA}
The bandwidth \( \hiso_{0} \) can be selected as the smallest value ensuring
\begin{EQA}[c]
	\frac{1}{J}\sum_{j=1}^{J} \sumi \Kern\bigl( \hiso^{-2} \| \Xv_{i} - \xv_{j} \|^{2} \bigr)
%	=
%	\frac{1}{J}\sum_{j=1}^{J} \sumi \weight_{ij}
	\geq 
	\nmin
	\, .
\label{jhgiverkytde4w34etycj}
\end{EQA}
%
Anisotropic weights assume some prior information about the structure of the model, which is accumulated 
in the localizing tensor \( \TEDR \).
Namely, for two values \( \hiso \) and \( \haniso \) and a current guess \( \betav \), anisotropic weights \( \weight_{ij} \)
correspond to the tensor 
\begin{EQA}[rcl]
	\TEDR^{2}
	&=&
	\hiso^{-2} \Id_{\dimd} + \haniso^{-2} \betav \betav^{\T}
\label{iufhvu8kjbmnll475edf}
\end{EQA}
leading to 
\begin{EQA}[c]
	\weight_{ij}
	=
	\Kern\bigl( \| \TEDR (\Xv_{i} - \xv_{j}) \|^{2} \bigr)
	=
	\Kern\bigl( \hiso^{-2} \| \Xv_{i} - \xv_{j} \|^{2} + \haniso^{-2} \langle \Xv_{i} - \xv_{j}, \betav \rangle^{2} \bigr)
	\, .
\end{EQA}
The ratio \( \hiso/\haniso \) describes the anisotropy strength.
Large values of \( \hiso/\haniso \) correspond to strong localization in direction \( \betav \)
and a mild localization in orthogonal directions. 
During iteration, the procedure increases the anisotropy parameter \( \hiso/\haniso \).
For fixed \( \haniso \) and \( \betav \), the bandwidth \( \hiso \) is determined by the condition
\begin{EQA}[c]
	\frac{1}{J} \sum_{j=1}^{J} \sumi 
		\Kern\bigl( \hiso^{-2} \| \Xv_{i} - \xv_{j} \|^{2} + \haniso^{-2} \langle \Xv_{i} - \xv_{j}, \betav \rangle^{2} \bigr)
%	=
%	\frac{1}{J}\sum_{j=1}^{J} \sumi \weight_{ij}
	\geq 
	\nmin
	\, .
\label{jhgiverkytde4w34etycja}
\end{EQA}
It is easy to see that a decay of \( \haniso \) leads to an increase of \( \hiso \).


\Subsection{Design of experiments and implementation}
Fix the values \( d \) and \( n \) such that \( n \geq 10 d \).
Also, fix the values \( \sigma_{\eps} \), \( \sigmaX \), \( q \in (0,1] \), and \( \rho \in [0,1) \).
Finally, fix a unit vector \( \betavs \) and a univariate function \( \fs(x) \); e.g \( f(x) = \sin(x) \).
\begin{mylist}

\item 
generate a correlated design \( \Xv_{i} = \sigmaX \bigl\{ \rho \gaussv_{0} + (1 - \rho) \gaussv_{i} \bigr\} \),
where \( \gaussv_{i} \) are i.i.d. standard gaussian in \( \R^{\dimd} \), \( i=0,1,\ldots,n \);
\item 
for each \( \Xv_{i} \), decide by Bernoulli experiment with positive probability \( q \), if this point is used for 
constructing the center \( \xv_{j} \) by \( \xv_{j} = \Xv_{i} + \sigmaX \gaussv_{j} \);
fix \( \JJJ = \#(\xv_{j}) \approx q n \);
\item 
generate errors \( \eps_{i} \sim \ND(0,\sigma_{\eps}^{2}) \) and compute \( Y_{i} = \fs(\Xv_{i}^{\T} \betavs) + \eps_{i} \);
\end{mylist}

\paragraph{Implementation. Tuning parameters}
For the procedure, we have to fix
\begin{mylist}

\item 
the value \( \nmin \gg 1 \); par defaut, \( \nmin = 10 \);
\item 
the factor \( a > 1 \); par defaut, \( a = \sqrt{2} \);
\item 
the kernel \( \Kern \); par defaut, Epanechnikov kernel \( \Kern(t) = (1-t^{2})_{+} \);
\item 
the Lagrange penalizing constant \( \lambda \), par defaut, \( \lambda = \nmin \);
\end{mylist}

	\, ;

\begin{EQA}[rcl]

\label{iufhvu8kjbmnll475edfk}

\item

Find \( \prmtv^{(k)} = \{ \betav^{(k)},(\fcn_{j}^{(k)},\fcl_{j}^{(k)}) \} \) as minimizer of

\Section{Multi-index model}

\Subsection{Structural adaptation}
\label{SEDRsa}
As already mentioned, the structural multi-index information about the model
can be recoved from the gradient vectors \( \nabla \fs(\xv) \).
More precisely, under \eqref{ucjcvjuvyvyf6r6rwjdy}, the matrix
%by orthogonality of \( \betav_{1}, \ldots,\betav_{m} \)
\begin{EQA}[c]
	\JEDR
	\eqdef
	\sum_{j} \nabla \fs(\xv_{j}) \, \nabla \fs(\xv_{j})^{\T}
	=
	\sum_{j} \PEDR \fclv_{j} \, \bigl( \PEDR \fclv_{j} \bigr)^{\T}

\end{EQA}
accumulates the structural information about the model.
The sum
\begin{EQA}
	\JEDR_{0}

	\sum_{j} (\BEDR_{0}^{\T} \fclv_{j}^{(0)}) \, (\BEDR_{0}^{\T} \fclv_{j}^{(0)})^{\T}

\end{EQA}
with \( \hiso > \haniso \).

Also fix \( \PEDR_{0} \) using the eigenvalue decomposition of \( \JEDR_{0} \):
\begin{EQA}[c]

\Section{Manifold learning}

Consider a regression \( Y_{i} = \fs(\Xv_{i}) + \eps_{i} \) with a high-dimensional \( \Xv_{i} \).
The manifold hypothesis assumes that the function \( \fs(\cdot) \) has a nearly multi-index structure in a local vicinity 
of each point \( \xv \in \R^{\dimd} \).
The goal is to extend the multi-index procedure from Section~\ref{Smultind} to this situation 
by allowing the mapping \( \PEDR \) to depend on the location \( \xv \) with a strong penalty on their variability.

Suppose that the procedure has been initialized and a current solution is given in terms of 
Jacobians \( \JEDR_{j}^{(k-1)} = \JEDR^{(k-1)}(\xv_{j}) \) and their SVD representation
\begin{EQA}[c]

	\, ,

\Subsection{Single-index}

The single-index procedure from Section~\ref{Ssingind} recovers 
as the single-index vector \( \betav \), as the coefficients \( \fcn_{j}, \fcl_{j} \)

The average derivative approach focuses on estimation of directional derivatives. 
We use the notations of Section~\ref{Ssingind}.

	\, ,

\Subsection{Multiple index procedure}
\label{Saverdermi}
An extension to a multiple index procedure is straightforward.
Consider the e.d.r. regression problem
\begin{EQA}
	Y_{i}
	&=&
	\fs(\PEDR \Xv_{i}) + \eps_{i}
\label{YifXieiedrAVE}
\end{EQA}
where \( \Xv_{i} \in \R^{\dimd} \) for large \( \dimd \), 
\( \PEDR \) is an unknown element from the set \( \PROJ_{\dimd,\dimm} \) 
of orthogonal mappings from 
from \( \R^{\dimd} \) to \( \R^{\dimm} \) for \( \dimm \ll \dimd \), and
\( \fs \) is a function on \( \R^{\dimm} \).
The mapping \( \PEDR \) can be written as 
\begin{EQA}
	\PEDR \xv
	&=&
	(\betav_{1}^{\T} \xv) \betav_{1} + \ldots + (\betav_{\dimm}^{\T} \xv) \betav_{\dimm}
	\, ,
\label{ydgenb8jejAVE}
\end{EQA}
where \( \betav_{m}^{\T} \) are the orthonormal rows of \( \PEDR \).

Suppose to be given a tensor \( \TEDR \), let the weights \( \weight_{ij} \) be defined as in \eqref{s7cuqkdwywdhbjwbfjwbdj}
for all \( i=1,\ldots,n \) and \( j \leq \JJJ \).
The sums \( Z_{j} \) from \eqref{juu8hjfioee433erhAVE} can be expanded as
\begin{EQA}[rcl]
	\E Z_{j}
	& \approx &
	\sumi \biggl\{ \fcn_{j} + \sum_{\mm=1}^{\dimm} \fcl_{j,\mm} \, \betav_{\mm}^{\T} (\Xv_{i} - \xv_{j}) \biggr\} \, \langle \Xv_{i} - \Sv_{j} \, ,\uv_{j} \rangle \, \weight_{ij}
	\\
	&=&
	\sum_{\mm=1}^{\dimm} \sumi \fcl_{j,\mm} \, \betav_{\mm}^{\T} (\Xv_{i} - \Sv_{j}) \, \langle \Xv_{i} - \Sv_{j} \, ,\uv_{j} \rangle \, \weight_{ij}
%	=
%	\fcl_{j} \, \uv_{j}^{\T} \KSme_{j} \betav
	=
	\Uv_{j}^{\T} \sum_{\mm=1}^{\dimm} \fcl_{j,\mm} \, \betav_{\mm}
	=
	\fclv_{j}^{\T} \PEDR \, \Uv_{j}
	\, .
\end{EQA}
Therefore, each step of the procedure can be reduced to optimization
\begin{EQA}[c]
	\sum_{j} \bigl\| \Ima_{j} - \fclv_{j}^{\T} \BEDR \, \Uv_{j} \bigr\|^{2} + \lambda \| \BEDR - \PEDR_{0} \|_{\Fr}^{2}
	\to
	\min_{\BEDR,(\fclv_{j})}
	\, ,
\label{ikdfjoiefjoieuAVEmi}
\end{EQA}
where \( \BEDR \in \R^{\dimm \times \dimd} \), \( \fclv_{j} \in \R^{\dimm} \).
This problem can be effectively solved by alternating optimization.
The obtained solution can be used for structural adaptation as described in Section~\ref{SEDRsa}.
